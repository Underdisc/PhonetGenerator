85 void Options::parse_options(int argc, char ** argv)
86 {
87  bool error = false;
88  int opt;
89  while(1)
90  {
91  int opt_index = 0;
92  static struct option long_options[] =
93  {
94  {"minimum-length", required_argument, NULL, 'n'},
95  {"maximum-length", required_argument, NULL, 'x'},
96  {"words", required_argument, NULL, 'w'},
97  {"spellings", required_argument, NULL, 's'},
98  {"pronunciation", no_argument, NULL, 'p'},
99  {"seed", required_argument, NULL, 'r'},
100  {"file", required_argument, NULL, 'f'},
101  {"help", no_argument, NULL, 'h'}
102  };
103 
104  opt = getopt_long(argc, argv, ":n:x:w:s:ps:r:f:h", long_options, &opt_index);
105  if(opt == -1)
106  break;
107  switch (opt)
108  {
109  case 'n': m_min_length = atoi(optarg); break;
110  case 'x': m_max_length = atoi(optarg); break;
111  case 'w': m_num_words = atoi(optarg); break;
112  case 's': m_num_spellings = atoi(optarg); break;
113  case 'p': m_pronunciation = true; break;
114  case 'r': m_seed = (unsigned)atoi(optarg); break;
115  case 'f': m_phoneme_file = optarg; break;
116  case 'h': m_help = true; break;
117  case '?': std::cout << "The provided option [" << (char)optopt
118  << "] is not an option." << std::endl;
119  error = true; break;
120  case ':': std::cout << "The provided option [" << (char)optopt
121  << "] requires an argument." << std::endl;
122  error = true; break;
123  }
124  }
125  if(error)
126  std::cout << "Use -h or --help to see all options and how they are used."
127  << std::endl;
128 }
