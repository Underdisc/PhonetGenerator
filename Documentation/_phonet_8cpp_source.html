106 void Phonet::generate(const PhonemePool & pool)
107 {
108  m_phonemes.clear();
109  size_t length = (rand() % (m_max_length - m_min_length + 1)) + m_min_length;
110  // picking first phoneme
111  if(length > 0)
112  {
113  m_phonemes.push_back(pool.get_phoneme());
114  --length;
115  }
116  else
117  return;
118  // picking remaining phonemes
119  while(length)
120  {
121  const Phoneme * next_phoneme = m_phonemes.back()->get_following_phoneme();
122  // make sure that the phoneme has at least one following phoneme
123  if(!next_phoneme)
124  {
125  length = 0;
126  }
127  else
128  {
129  m_phonemes.push_back(m_phonemes.back()->get_following_phoneme());
130  --length;
131  }
132  }
133 }
